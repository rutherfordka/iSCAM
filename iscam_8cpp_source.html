1039  NOTES:

1041  1) Logistic selectivity with 2 parameters.

1043  and the last two age-classes are assumed to have the same selectivity.
1044  3) A reduced age-specific parameter set based on a bicubic spline.

1046  5) Time varying bicubic spline (2d version).
1047  6) Fixed logistic.
1048  7) Logistic selectivity based on relative changes in mean weight at age
1049  8) Time varying selectivity based on logistic with deviations in
1050  weights at age (3 estimated parameters).
1051  11) Logistic selectivity with 2 parameters based on mean length.
1052  12) Length-based selectivity using cubic spline interpolation.
1053  - The bicubic_spline function is located in stats.cxx library.
1054  TODO list:
1055  [*] add an option for length-based selectivity. Use inverse of
1056  allometric relationship w_a = a*l_a^b; to get mean length-at-age from

1708 void model_parameters::calcStockRecruitment()
1709 {

1712  Purpose:

1714  relationship that is ultimately used in determining MSY-based reference
1715  points. The objective of this function is to determine the appropriate
1716  Ro, Bo and steepness values of either the Beverton-Holt or Ricker Stock-
1717  Recruitment Model:
1718  Beverton-Holt Model
1719  Rt=k*Ro*St/(Bo+(k-1)*St)*exp(delta-0.5*tau*tau)
1720  Ricker Model
1721  Rt=so*St*exp(-beta*St)*exp(delta-0.5*tau*tau)
1722  The definition of a stock is based on group only. At this point, spawning biomass
1723  from all areas for a given group is the assumed stock, and the resulting
1724  recruitment is compared with the estimated recruits|group for all areas.
1725  Author: Steven Martell
1726  Arguments:
1727  None
1728  NOTES:
1729  Psuedocode:
1730  -1) Get average natural mortality rate at age.
1731  -2) Calculate survivorship to time of spawning.
1732  -3) Calculate unfished spawning biomass per recruit.
1733  -4) Compute spawning biomass vector & substract roe fishery
1734  -5) Project spawning biomass to nyr+1 under natural mortality.
1735  -6) Calculate stock recruitment parameters (so, beta);
1736  -7) Calculate predicted recruitment
1737  -8) Compute residuals from estimated recruitments.
1738  TODO list:
1739  [] - Change step 3 to be a weighted average of spawning biomass per recruit by area.
1740  [] - Increase dimensionality of ro, sbo, so, beta, and steepness to ngroup
1741  [ ] - Add autocorrelation in recruitment residuals with parameter \gamma_r.

1744  rt.initialize();
1745  sbt.initialize();
1746  delta.initialize();
1747  dvariable phib;//,so,beta;
1748  dvector fa(sage,nage);
1749  dvar_vector stmp(sage,nage);
1750  dvar_vector ma(sage,nage);

1753  dvar_vector lw(sage,nage);
1754  for(g=1;g<=ngroup;g++)
1755  {
1756  lx.initialize();
1757  lw.initialize();
1758  lx(sage) = 1.0;
1759  lw(sage) = 1.0;
1760  phib = 0;
1761  for(f=1;f<=narea;f++)
1762  {
1763  for(h=1;h<=nsex;h++)
1764  {
1765  ig = pntr_ags(f,g,h);
1766  // | Step 1. average natural mortality rate at age.
1767  // | Step 2. calculate survivorship
1768  for(j=sage;j<=nage;j++)
1769  {
1770  ma(j) = mean(trans(M(ig))(j));
1771  fa(j) = mean( trans(d3_wt_mat(ig))(j) );
1772  if(j > sage)
1773  {
1774  lx(j) = lx(j-1) * mfexp(-ma(j-1));
1775  }
1776  lw(j) = lx(j) * mfexp(-ma(j)*d_iscamCntrl(13));
1777  }
1778  lx(nage) /= 1.0 - mfexp(-ma(nage));
1779  lw(nage) /= 1.0 - mfexp(-ma(nage));
1780  // | Step 3. calculate average spawing biomass per recruit.
1781  phib += 1./(narea*nsex) * lw*fa;
1782  // | Step 4. compute spawning biomass at time of spawning.
1783  for(i=syr;i<=nyr;i++)
1784  {
1785  stmp = mfexp(-Z(ig)(i)*d_iscamCntrl(13));
1786  sbt(g,i) += elem_prod(N(ig)(i),d3_wt_mat(ig)(i)) * stmp;
1787  }
1788  // | Step 5. spawning biomass projection under natural mortality only.
1789  stmp = mfexp(-M(ig)(nyr)*d_iscamCntrl(13));
1790  sbt(g,nyr+1) += elem_prod(N(ig)(nyr),d3_wt_mat(ig)(i)) * stmp;
1791  }
1792  // | Estimated recruits
1793  ih = pntr_ag(f,g);
1794  rt(g) += mfexp(log_rt(ih)(syr+sage,nyr));
1795  }
1796  // | Step 6. calculate stock recruitment parameters (so, beta, sbo);
1797  so(g) = kappa(g)/phib;
1798  sbo(g) = ro(g) * phib;
1799  // | Step 7. calculate predicted recruitment.
1800  dvar_vector tmp_st = sbt(g)(syr,nyr-sage).shift(syr+sage);
1801  switch(int(d_iscamCntrl(2)))
1802  {
1803  case 1: // | Beverton Holt model
1804  beta(g) = (kappa(g)-1.)/sbo(g);
1805  tmp_rt = elem_div(so(g)*tmp_st,1.+beta(g)*tmp_st);
1806  break;
1807  case 2: // | Ricker model
1808  beta(g) = log(kappa(g))/sbo(g);
1809  tmp_rt = elem_prod(so(g)*tmp_st,exp(-beta(g)*tmp_st));
1810  break;
1811  }
1812  // | Step 8. // residuals in stock-recruitment curve with gamma_r = 0
1813  delta(g) = log(rt(g))-log(tmp_rt)+0.5*tau*tau;
1814  // Autocorrelation in recruitment residuals.
1815  // if gamma_r > 0 then
1816  if( active(gamma_r) )
1817  {
1818  int byr = syr+sage+1;
1819  delta(g)(byr,nyr) = log(rt(g)(byr,nyr))
1820  - (1.0-gamma_r)*log(tmp_rt(byr,nyr))
1821  - gamma_r*log(++rt(g)(byr-1,nyr-1))
1822  + 0.5*tau*tau;
1823  }
1824  }
1825  if(verbose)cout<<"**** Ok after calcStockRecruitment ****"<<endl;
1826  }
