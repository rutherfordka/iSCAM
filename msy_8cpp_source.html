121 // Calculate msy value given an allocation for each gear type.
122 void Msy::get_fmsy(dvector& fe, dvector& ak)
123 {
124  /*
125  This algorithm iteratively solves for a vector fe that will
126  maximize the sum of catches over all fleets where the catch
127  is allocated (ak, where sum ak=1) to each fleet.
128 
129  How it works:
130  - Run the model with all fleets having the same Fe
131  - Then calculate fmultiplier (lambda) as the ratio
132  allocation:yield/sum(yield) and correct Fe's down or up
133  such that the yield proportions matches the allocation.
134  - Run the equilibrium model again to get the derivative
135  information for the total yield and use Newton-Rhaphson to
136  update the estimate of fbar.
137  - Repeat above steps until derivative for the total yield
138  approaches zero (TOL).
139 
140  */
141  // int i;
142  int n = size_count(fe);
143  int iter =1;
144  ak = ak/sum(ak);
145  double fbar = mean(fe);
146  dvector lambda = ak/mean(ak);
147  dvector fk = elem_prod(fe,lambda);
148  dvector p(1,n);
149  double x1, x2;
150  x1 = 1.0e-5;
151  x2 = 3.0e02;
152  // Spawning biomass per recruit for unfished conditions
153  calc_phie(m_dM,m_dFa);
154 
155  do
156  {
157  // Run the model with constant fk = fbar to get
158  // lambda multipliers.
159  fk = fbar;
160  calcEquilibrium(fk);
161  lambda = elem_div(ak,m_ye/sum(m_ye));
162 
163  // Run the model with fk = lambda*fbar to get
164  // derivatives of the total catch equation to update fbar;
165  fk = fbar*lambda;
166  calcEquilibrium(fk);
167  //cout<<iter<<" fbar "<<fbar<<" dYe "<<fabs(m_dYe)<<" fk "<<fk<<endl;
168 
169  fbar = fbar - m_dYe/m_d2Ye;
170  iter++;
171 
172  // Check boundary conditions and reduce step size if necessary.
173  if( (x1-fbar)*(fbar-x2)<0.0 )
174  {
175  cout<<"get_fmsy:: reducing Newton-step"<<endl;
176  fbar += 0.98 * m_dYe/m_d2Ye;
177  //exit(1);
178  }
179 
180  }
181  while ( sqrt(square(m_dYe)) >TOL && iter < MAXITER );
182 
183  if( iter == MAXITER ) m_FAIL = true;
184 
185  fe = fk;
186 
187  m_fmsy = fe;
188  m_msy = m_ye;
189  m_bmsy = m_be;
190  m_rmsy = m_re;
191  m_spr_msy = m_spr;
192 
193 }

195 // calculate survivorship under fished conditions
196 void Msy::calc_equilibrium(const dvector& fe)
197 {
198  /* TO BE Deprecated in iSCAM */
199  /*
200 
201  This is the main engine behind the Msy class.
202  Using the private member variables, calculate the equilibrium yield
203  for a given fishing mortality rate vector (fe). Also, compute the
204  derivative information such that this info can be used to compute
205  the corresponding MSY-based reference points (Fmsy, MSY, Bmsy).
206 
207  There are two different survivorship calculations inside this routine:
208  lz is the survivorship of the total numbers, and lw is the survivorship
209  upto the time of spawning. The spawning biomass per recruit (phif) depends
210  on the survivorship up to the time of spawning.
211  */
212 
213  // Indexes for dimensions
214  int j,k;
215  int sage,nage,ngear;
216  sage = m_wa.indexmin();
217  nage = m_wa.indexmax();
218  ngear = m_V.rowmax();
219 
220 
221  // Survivorship for fished conditions.
222  dvector lz(sage,nage);
223  dvector lw(sage,nage);
224  dmatrix dlz(1,ngear,sage,nage);
225  dmatrix d2lz(1,ngear,sage,nage);
226  dmatrix dlw(1,ngear,sage,nage);
227  dmatrix d2lw(1,ngear,sage,nage);
228  dlz.initialize();
229  d2lz.initialize();
230  dlw.initialize();
231  d2lw.initialize();
232 
233  dvector za = m_M + fe * m_V;
234  dvector pza = m_rho*za;
235  dvector sa = exp(-za);
236  dvector psa = exp(-pza);
237  dvector oa = 1.-sa;
238  dvector poa = 1.-elem_prod(sa,psa);
239  dmatrix qa(1,ngear,sage,nage);
240 
241  for(k=1;k<=ngear;k++)
242  {
243  qa(k) = elem_div(elem_prod(elem_prod(m_V(k),m_wa),oa),za);
244  dlz(k,sage) = 0;
245  d2lz(k,sage) = 0;
246  dlw(k,sage) = -psa(sage)*m_rho*m_V(k)(sage);
247  d2lw(k,sage) = psa(sage)*square(m_rho)*square(m_V(k)(sage));
248  }
249 
250  lz(sage) = 1.0;
251  lw(sage) = 1.0 * psa(sage);
252  for(j=sage+1; j<=nage; j++)
253  {
254  lz(j) = lz(j-1) * sa(j-1);
255  lw(j) = lz(j) * psa(j);
256  if( j==nage )
257  {
258  lz(j) = lz(j)/oa(j);
259  lw(j) = lz(j-1)*sa(j-1)*psa(j)/oa(j);
260  }
261 
262  for(k=1; k<=ngear; k++)
263  {
264  // derivatives for survivorship
265  dlz(k)(j) = sa(j-1) * ( dlz(k)(j-1) - lz(j-1)*m_V(k)(j-1) );
266  d2lz(k)(j) = sa(j-1) * ( d2lz(k)(j-1)+ lz(j-1)*m_V(k)(j-1)*m_V(k)(j-1) );
267 
268  // derivatives for spawning survivorship
269  dlw(k)(j) = -lz(j)*m_rho*m_V(k)(j)*psa(j);
270  d2lw(k)(j) = lz(j)*square(m_rho)*square(m_V(k)(j))*psa(j);
271 
272  if( j==nage ) // + group derivatives
273  {
274  dlz(k)(j) = dlz(k)(j)/oa(j) - lz(j-1)*sa(j-1)*m_V(k)(j)*sa(j)/square(oa(j));
275 
276  dlw(k)(j) = -lz(j-1)*sa(j-1)*m_rho*m_V(k)(j)/oa(j)
277  - lz(j-1)*psa(j)*m_V(k)(j)*sa(j)/square(oa(j));
278 
279  double V1 = m_V(k)(j-1);
280  double V2 = m_V(k)(j);
281  double oa2 = oa(j)*oa(j);
282 
283  d2lz(k)(j) = d2lz(k)(j)/oa(j)
284  + 2*lz(j-1)*V1*sa(j-1)*V2*sa(j)/oa2
285  + 2*lz(j-1)*sa(j-1)*V2*V2*sa(j)*sa(j)/(oa(j)*oa2)
286  + lz(j-1)*sa(j-1)*V2*V2*sa(j)/oa2;
287 
288  d2lw(k)(j) = lz(j-1)*square(m_rho)*square(V2)*psa(j)/oa(j)
289  + 2*lz(j-1)*m_rho*square(V2)*psa(j)*sa(j)/oa2
290  + 2*lz(j-1)*psa(j)*square(V2)*square(sa(j))/(oa(j)*oa2)
291  + lz(j-1)*psa(j)*square(V2)*sa(j)/oa2;
292  }
293  }// gear
294 
295  }// age
296 
297  // Incidence functions and associated derivatives
298  double ro = m_ro;
299  double kappa = 4.0*m_h/(1.0-m_h); // Beverton-Holt model
300  double km1 = kappa-1.0;
301  //double phif = lz * m_fa;
302  double phif = lw * m_fa;
303  double phif2 = phif*phif;
304  dvector dphif(1,ngear);
305  dvector d2phif(1,ngear);
306  dvector phiq(1,ngear);
307  dvector dphiq(1,ngear);
308  dvector d2phiq(1,ngear);
309  dvector dre(1,ngear);
310  dvector d2re(1,ngear);
311  dvector t1(sage,nage);
312 
313  for(k=1; k<=ngear; k++)
314  {
315  dphif(k) = dlz(k) * m_fa;
316  d2phif(k) = d2lz(k) * m_fa;
317  //dphif(k) = dlw(k) * m_fa;
318  //d2phif(k) = d2lw(k) * m_fa;
319 
320  // per recruit yield
321  phiq(k) = lz * qa(k);
322  if(ngear==1)
323  {
324  // dphiq = wa*oa*va*dlz/za + lz*wa*va^2*sa/za - lz*wa*va^2*oa/za^2
325  t1 = elem_div(elem_prod(elem_prod(lz,m_wa),square(m_V(k))),za);
326  }
327  else
328  {
329  // dphiq = wa*oa*va*dlz/za + lz*wa*va*sa/za - lz*wa*va*oa/za^2
330  t1 = elem_div(elem_prod(elem_prod(lz,m_wa),m_V(k)),za);
331  }
332  dvector t0 = elem_div(oa,za);
333  dvector t3 = sa-t0;
334  dphiq(k) = qa(k)*dlz(k) + t1 * t3;
335 
336  // 2nd derivative for per recruit yield (nasty)
337  dvector t2 = 2. * dlz(k);
338  dvector V2 = elem_prod(m_V(k),m_V(k));
339  dvector t5 = elem_div(elem_prod(m_wa,V2),za);
340  dvector t7 = elem_div(m_V(k),za);
341  dvector t9 = elem_prod(t5,sa);
342  dvector t11 = elem_prod(t5,t0);
343  dvector t13 = elem_prod(lz,t5);
344  dvector t14 = elem_prod(m_V(k),sa);
345  dvector t15 = elem_prod(t7,sa);
346  dvector t17 = elem_prod(m_V(k),t0);
347  dvector t18 = elem_div(t17,za);
348  d2phiq(k) = d2lz(k)*qa(k) + t2*t9 - t2*t11 - t13*t14 -2.*t13*t15 + 2.*t13*t18;
349 
350 
351  // 1st & 2nd partial derivatives for recruitment
352  dre(k) = ro*m_phie*dphif(k)/(phif2*km1);
353  d2re(k) = -2.*ro*m_phie*dphif(k)*dphif(k)/(phif2*phif*km1)
354  + ro*m_phie*d2phif(k)/(phif2*km1);
355  }
356 
357  // Equilibrium calculations
358  double re;
359  dvector ye(1,ngear);
360  dvector fstp(1,ngear);
361  dvector dye(1,ngear);
362  dmatrix d2ye(1,ngear,1,ngear);
363  dmatrix invJ(1,ngear,1,ngear);
364 
365  re = ro*(kappa-m_phie/phif) / km1;
366  ye = re*elem_prod(fe,phiq);
367 
368  // Aug 6, 2013, found a bug in my linear algebra.
369  // dye = re*phiq + elem_prod(fe,elem_prod(phiq,dre)) + elem_prod(fe,re*dphiq);
370  dye = re*phiq + elem_prod(fe,phiq)*dre + (fe*re)*dphiq;
371 
372  // Caclculate Jacobian matrix (2nd derivative of the catch equation)
373  for(j=1; j<=ngear; j++)
374  {
375  for(k=1; k<=ngear; k++)
376  {
377  d2ye(k)(j) = fe(j)*phiq(j)*d2re(k) + 2.*fe(j)*dre(k)*dphiq(k) + fe(j)*re*d2phiq(k);
378  if(k == j)
379  {
380  d2ye(j)(k) += 2.*dre(j)*phiq(j)+2.*re*dphiq(j);
381  }
382  }
383  }
384  // Newton-Raphson step.
385  invJ = -inv(d2ye);
386  fstp = invJ * dye;
387 
388  // Set private members
389  m_p = fstp;
390  m_ye = ye;
391  m_re = re;
392  m_be = re*phif;
393  m_bi = re*(lz*m_fa);
394  m_spr = phif/m_phie;
395  m_dYe = sum(dye);
396  m_d2Ye = sum(diagonal(d2ye));
397  m_g = diagonal(d2ye); //Gradient vector
398  m_f = dye; //Value of the function to minimize
399 
400  // cout<<"OB1 fe "<<fe<<" ye "<<ye<<" dye "<<dye<<endl;
401 }

628  if(ngear>=1) // was (if ngear==1), changed during debugging of nfleet>1
629  {

631  t1 = elem_div(elem_prod(elem_prod(lz_m(h),m_dWa(h)),square(m_d3_V(h)(k))),za_m(h));
632  }

637  }

640  dphiq(k) += qa_m(h)(k)*dlz_m(h)(k) + t1 * t3;
641

643  dvector t2 = 2. * dlz_m(h)(k);
644  dvector V2 = elem_prod(m_d3_V(h)(k),m_d3_V(h)(k));

651  dvector t15 = elem_prod(t7,sa_m(h));
652  dvector t17 = elem_prod(m_d3_V(h)(k),t0);
653  dvector t18 = elem_div(t17,za_m(h));
654  d2phiq(k) += d2lz_m(h)(k)*qa_m(h)(k)
655  + t2*t9 - t2*t11 - t13*t14 -2.*t13*t15
656  + 2.*t13*t18;
657  } // gear
658  } // ngrp

661  {

665  + ro*m_phie*d2phif(k)/(phif2*km1);

679  dye = re*phiq + elem_prod(fe,phiq)*dre + (fe*re)*dphiq;

692  {

758 void Msy::calc_phie()
759 {
760  int i,sage,nage;
761  sage = m_fa.indexmin();
762  nage = m_fa.indexmax();
763  dvector lx(sage,nage);
764  for(i=sage; i<=nage; i++)
765  {
766  lx(i) = exp( -m_M*(i-sage) -m_rho*m_M );
767  if(i==nage)
768  lx(i) /= 1.0 - exp( -m_M );
769  }

792 void Msy::calc_phie(const dmatrix& _m, const dmatrix& _fa)
793 {
794  int i,j,sage,nage;
795  m_dM = _m;
796  m_dFa = _fa;
797  int r1 = m_dM.rowmin();
798  int r2 = m_dM.rowmax();
799  sage = m_dFa.colmin();
800  nage = m_dFa.colmax();
801  dvector lx(sage,nage);
802  m_phie = 0;
803  for(i=m_dM.rowmin(); i<=m_dM.rowmax(); i++)
804  {
805  lx.initialize();
806  for( j = sage; j <= nage; j++ )
807  {
808  lx(j) = exp( -m_dM(i,j)*(j-sage) - m_rho*m_dM(i,j) );
809  if( j==nage )
810  lx(j) /= 1.0 - exp( -m_dM(i,j) );
811  }
812 
813  m_phie += 1./(r2-r1+1)*lx * m_dFa(i);
814  }
815

817  m_bo = m_ro * m_phie;
